interface PersonalInfo {

              <tr className="border-b border-primary/8">
                <th className="text-left px-5 py-3 text-xs font-semibold text-primary/40 uppercase tracking-wider">Tenant</th>
                <th className="text-left px-5 py-3 text-xs font-semibold text-primary/40 uppercase tracking-wider hidden md:table-cell">Stay Period</th>
                <th className="text-left px-5 py-3 text-xs font-semibold text-primary/40 uppercase tracking-wider">Status</th>
                <th className="text-left px-5 py-3 text-xs font-semibold text-primary/40 uppercase tracking-wider hidden sm:table-cell">ID Type</th>
                <th className="text-left px-5 py-3 text-xs font-semibold text-primary/40 uppercase tracking-wider">Action</th>
              </tr>
            </thead>
            <tbody>
              {bookingData.slice(0, 5).map((b) => (
                <tr key={b.id} className="border-b border-primary/5 last:border-0 hover:bg-primary/3 transition-colors">
                  <td className="px-5 py-3.5">
                    <p className="font-semibold text-primary">{b.fullName}</p>
                    <p className="text-[10px] text-primary/40 font-mono uppercase truncate max-w-[80px]">{b.id}</p>
                  </td>
                  <td className="px-5 py-3.5 text-primary/60 hidden md:table-cell">
                    {b.entryDate} - {b.exitDate}
                  </td>
                  <td className="px-5 py-3.5"><StatusBadge status={b.bookingStatus} /></td>
                  <td className="px-5 py-3.5 font-semibold text-primary hidden sm:table-cell capitalize">{b.idType}</td>
                  <td className="px-5 py-3.5">
                    <button
                      onClick={() => setSelectedBooking(b)}
                      className="text-xs font-bold text-accent hover:text-accent-light transition-colors"
                    >
                      View Details
                    </button>
                  </td>
                </tr>
              ))}
              {bookingData.length === 0 && (
                 <tr><td colSpan={5} className="py-10 text-center text-primary/40 text-xs">No bookings found.</td></tr>
              )}
            </tbody>
          </table>
        </div>
      </div>

      {/* Recent messages - Keep original UI */}
      <div className="bg-cream rounded-2xl border border-primary/8 overflow-hidden">
        <div className="px-5 py-4 border-b border-primary/8 flex items-center justify-between">
          <h2 className="font-display font-bold text-primary text-base">Recent Messages</h2>
          <button onClick={() => onTabChange('messages')} className="text-xs font-semibold text-accent hover:text-accent-light transition-colors">View all →</button>
        </div>
        <div className="divide-y divide-primary/5">
          {messages.slice(0, 2).map((msg) => (
            <div
              key={msg.id}
              className={`px-5 py-4 flex items-start gap-3 cursor-pointer hover:bg-primary/3 transition-colors ${!msg.read ? 'bg-accent/3' : ''}`}
              onClick={() => setSelectedMessage(msg)}
            >
              <div className="w-8 h-8 rounded-full bg-primary/10 flex items-center justify-center flex-shrink-0 text-xs font-bold text-primary">
                {msg.name.charAt(0)}
              </div>
              <div className="flex-1 min-w-0">
                <div className="flex items-center gap-2">
                  <p className="font-semibold text-primary text-sm">{msg.name}</p>
                  {!msg.read && <span className="w-1.5 h-1.5 rounded-full bg-accent flex-shrink-0" />}
                </div>
                <p className="text-xs text-primary/50 font-medium">{msg.subject}</p>
                <p className="text-xs text-primary/40 mt-0.5 truncate">{msg.message}</p>
              </div>
              <span className="text-xs text-primary/30 flex-shrink-0">{msg.date}</span>
            </div>
          ))}
        </div>
      </div>

      {/* Booking Detail Modal - Updated for IBookingData */}
      <Modal isOpen={!!selectedBooking} onClose={() => setSelectedBooking(null)} title="Verify Booking" size="lg">
        {selectedBooking && (
          <div className="space-y-5">
            <div className="flex items-center justify-between border-b border-primary/5 pb-4">
              <div>
                <p className="font-display font-bold text-primary text-xl">{selectedBooking.fullName}</p>
                <p className="text-sm text-primary/50">{selectedBooking.email} • {selectedBooking.phone}</p>
              </div>
              <div className="text-right">
                <StatusBadge status={selectedBooking.bookingStatus} />
                <p className="text-[10px] text-primary/30 mt-1 uppercase font-mono">{selectedBooking.id}</p>
              </div>
            </div>

            {/* Document Verification Section */}
            <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
              <div className="space-y-2">
                <p className="text-[10px] font-bold text-primary/40 uppercase tracking-widest">Identity ({selectedBooking.idType})</p>
                <div className="aspect-video bg-primary/5 rounded-xl border border-primary/8 overflow-hidden group relative">
                    <img 
                      src={typeof selectedBooking.idDocumentImage === 'string' ? selectedBooking.idDocumentImage : '/placeholder-id.png'} 
                      alt="ID Document" 
                      className="w-full h-full object-cover"
                    />
                </div>
              </div>
              <div className="space-y-2">
                <p className="text-[10px] font-bold text-primary/40 uppercase tracking-widest">Payment Proof</p>
                <div className="aspect-video bg-primary/5 rounded-xl border border-primary/8 overflow-hidden group relative">
                    <img 
                      src={typeof selectedBooking.paymentProofImage === 'string' ? selectedBooking.paymentProofImage : '/placeholder-payment.png'} 
                      alt="Payment" 
                      className="w-full h-full object-cover"
                    />
                </div>
              </div>
            </div>

            <div className="grid grid-cols-2 sm:grid-cols-4 gap-3">
              {[
                { label: 'Room Pref', value: selectedBooking.roomPreference },
                { label: 'Occupants', value: selectedBooking.numberOfOccupants },
                { label: 'Check-In', value: selectedBooking.entryDate },
                { label: 'Check-Out', value: selectedBooking.exitDate },
              ].map(({ label, value }) => (
                <div key={label} className="bg-primary/4 rounded-xl p-3">
                  <p className="text-[10px] text-primary/40 font-semibold uppercase tracking-wider mb-1">{label}</p>
                  <p className="text-sm font-semibold text-primary capitalize">{value}</p>
                </div>
              ))}
            </div>

            <div className="flex gap-2 pt-2">
              <button 
                onClick={() => handleUpdateStatus(selectedBooking.id!, PaymentStatus.VERIFIED)}
                className="flex-1 py-3 px-4 bg-accent text-cream text-sm font-bold rounded-xl hover:opacity-90 transition-all shadow-sm shadow-accent/20"
              >
                Approve Payment
              </button>
              <button 
                onClick={() => dispatch(adminDeleteBooking(selectedBooking.id!)).then(() => setSelectedBooking(null))}
                className="flex-1 py-3 px-4 border border-red-200 text-red-500 text-sm font-bold rounded-xl hover:bg-red-50 transition-all"
              >
                Reject / Delete
              </button>
            </div>
          </div>
        )}
      </Modal>

      {/* Message Detail Modal - Keep original */}
      <Modal isOpen={!!selectedMessage} onClose={() => setSelectedMessage(null)} title="Message" size="md">
        {selectedMessage && (
          <div className="space-y-4">
            <div className="flex items-start gap-3">
              <div className="w-10 h-10 rounded-full bg-primary/10 flex items-center justify-center flex-shrink-0 font-bold text-primary">{selectedMessage.name.charAt(0)}</div>
              <div>
                <p className="font-semibold text-primary">{selectedMessage.name}</p>
                <p className="text-xs text-primary/50">{selectedMessage.email}</p>
                <p className="text-xs text-primary/30 mt-0.5">{selectedMessage.date}</p>
              </div>
            </div>
            <div className="bg-primary/4 rounded-xl p-4">
              <p className="text-sm font-semibold text-primary mb-2">{selectedMessage.subject}</p>
              <p className="text-sm text-primary/70 leading-relaxed">{selectedMessage.message}</p>
            </div>
            <button onClick={() => setSelectedMessage(null)} className="w-full py-2.5 px-4 bg-primary text-cream text-sm font-semibold rounded-xl hover:bg-primary/85 transition-all">Reply via Email</button>
          </div>
        )}
      </Modal>
    </div>
  );
}
